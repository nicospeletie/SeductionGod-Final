             * Caches all necessary DOM elements for performance.

                this.elements = {

            /**

             * Main initialization method.

             */

            init() {

                document.addEventListener('DOMContentLoaded', () => {

                    this.initObservers();

                    this.initModal();

                    this.initScarcitySimulation();

                    ['scroll', 'click', 'touchstart'].forEach(evt => 

                        document.body.addEventListener(evt, () => this.initAudio(), { once: true })

                    );

                });

            },


            // --- CORE MODULES ---


            /**

             * Manages all language-related functionality.

             */

            initTranslationsAndEvents() {

                this.state.currentLang = localStorage.getItem('seductionGodLang') || 'en';

                this.setLanguage(this.state.currentLang);

                this.elements.translateBtn.addEventListener('click', () => {

                    const newLang = this.state.currentLang === 'el' ? 'en' : 'el';

                    this.setLanguage(newLang);

                });

            },


            /**

             * Sets the language and updates all translatable text.

             * @param {string} lang - The language to set ('en' or 'el').

             */

            setLanguage(lang) {

                this.state.currentLang = lang;

                document.documentElement.lang = lang;

                localStorage.setItem('seductionGodLang', lang);

                this.elements.translateBtn.textContent = lang === 'el' ? 'EN' : 'GR';

                this.elements.translatable.forEach(el => {

                    const key = el.getAttribute('data-translate');

                    if (this.translations[lang][key]) {

                        el.innerHTML = this.translations[lang][key];

                    }

                });

                this.updateScarcityBarText(); // Ensure bar text updates on lang change

            },


            /**

             * Initializes the audio context for sound effects.

             * This must be triggered by a user interaction.

             */

            initAudio() {

                if (this.state.audioStarted) return;

                Tone.start();

                this.state.synth = new Tone.Synth({

                    oscillator: { type: 'sine' },

                    envelope: { attack: 0.005, decay: 0.1, sustain: 0.3, release: 1 }

                }).toDestination();

                this.state.audioStarted = true;

            },

            

            /**

             * Initializes the daily 2-hour countdown timers.

             */

            initCountdown() {

                const now = new Date();

                const today = now.toDateString();

                const storedData = JSON.parse(localStorage.getItem('seductionGodOffer'));


                if (storedData && storedData.day === today) {

                    this.state.countdownEndTime = storedData.endTime;

                } else {

                    this.state.countdownEndTime = now.getTime() + 2 * 60 * 60 * 1000;

                    localStorage.setItem('seductionGodOffer', JSON.stringify({

                        endTime: this.state.countdownEndTime,

                        day: today

                    }));

                }

                

                // Start the efficient animation frame loop

                this.animationLoop();

            },


            /**

             * Efficient animation loop using requestAnimationFrame.

             * Updates timers every second without performance degradation.

             */

            animationLoop() {

                let lastUpdateTime = 0;

                const update = (currentTime) => {

                    if (currentTime - lastUpdateTime > 1000) { // Update every second

                        this.updateTimersDisplay();

                        lastUpdateTime = currentTime;

                    }

                    requestAnimationFrame(update);

                };

                requestAnimationFrame(update);

            },


            /**

             * Updates the text content of all countdown timer elements.

             */

            updateTimersDisplay() {

                if(!this.state.countdownEndTime) return;

                const now = new Date().getTime();

                const distance = this.state.countdownEndTime - now;


                if (distance < 0) {

                    this.elements.priceBoxes.forEach(box => {

                        box.innerHTML = `<div class="limited-time" style="font-size: 1.5rem; color: var(--price-red);" data-translate="offer_expired">${this.translations[this.state.currentLang].offer_expired}</div>`;

                    });

                    return;

                }


                const hours = Math.floor((distance % (1000 * 60 * 60 * 24)) / (1000 * 60 * 60));

                const minutes = Math.floor((distance % (1000 * 60 * 60)) / (1000 * 60));

                const seconds = Math.floor((distance % (1000 * 60)) / 1000);


                const fHours = hours < 10 ? "0" + hours : hours;

                const fMinutes = minutes < 10 ? "0" + minutes : minutes;

                const fSeconds = seconds < 10 ? "0" + seconds : seconds;

                const formattedTime = `${fHours}:${fMinutes}:${fSeconds}`;


                this.elements.countdownTimers.forEach(timer => {

                    timer.textContent = formattedTime;

                });

            },


            /**

             * Initializes all Intersection Observers for animations and triggers.

             */

            initObservers() {

                // For fade-in sections

                const fadeInObserver = new IntersectionObserver((entries) => {

                    entries.forEach(entry => {

                        if (entry.isIntersecting) {

                            entry.target.classList.add('is-visible');

                            fadeInObserver.unobserve(entry.target);

                        }

                    });

                }, { threshold: 0.1 });

                this.elements.sections.forEach(el => fadeInObserver.observe(el));


                // For the scarcity bar

                const scarcityBarObserver = new IntersectionObserver((entries) => {

                    if (entries[0].isIntersecting) {

                        this.elements.scarcityBar.classList.add('is-visible');

                        scarcityBarObserver.unobserve(entries[0].target);

                    }

                }, { threshold: 0.1 });

                scarcityBarObserver.observe(this.elements.interstitialSection);


                // For the toast notification

                const toastObserver = new IntersectionObserver((entries) => {

                    if (entries[0].isIntersecting) {

                        setTimeout(() => this.showLiveFeedToast(), 1000);

                        toastObserver.unobserve(entries[0].target);

                    }

                }, { threshold: 0.5 });

                toastObserver.observe(this.elements.testimonialsSection);

            },


            /**

             * Initializes the logic for the entry pop-up modal.

             */

            initModal() {

                if (!sessionStorage.getItem('modalShown')) {

                    setTimeout(() => {

                        this.elements.modal.classList.add('is-visible');

                        sessionStorage.setItem('modalShown', 'true');

                    }, 2500);

                }

                const hideModal = () => this.elements.modal.classList.remove('is-visible');

                this.elements.closeModalBtn.addEventListener('click', hideModal);

                this.elements.modal.addEventListener('click', (e) => {

                    if (e.target === this.elements.modal) hideModal();

                });

            },

            

            /**

             * Manages the logic for the scarcity toast notification.

             */

            showLiveFeedToast() {

                if (this.state.toastShown) return;


                const locations = ["Athens", "Thessaloniki", "Patras", "Nicosia", "Limassol", "Heraklion"];

                const messages = [

                    { key: 'toast_social_proof', location: true, icon: '🌍' },

                    { key: 'toast_scarcity', location: false, icon: '🔥' }

                ];

                const randomMessage = messages[Math.floor(Math.random() * messages.length)];

                let messageText = this.translations[this.state.currentLang][randomMessage.key];


                if (randomMessage.location) {

                    const randomLocation = locations[Math.floor(Math.random() * locations.length)];

                    messageText = messageText.replace('{{location}}', randomLocation);

                }

                

                this.elements.toast.innerHTML = `${randomMessage.icon} ${messageText}`;

                

                if (this.state.audioStarted) {

                    this.state.synth.triggerAttackRelease("G5", "8n");

                }

                this.elements.toast.classList.add('is-visible');

                this.state.toastShown = true;

                sessionStorage.setItem('toastShown', 'true');


                setTimeout(() => {

                    this.elements.toast.classList.remove('is-visible');

                }, 5000);

            },

            

            /**

             * Initializes and manages the scarcity bar simulation.

             */

            initScarcitySimulation() {

                let spots = sessionStorage.getItem('spotsLeft');

                if (spots === null) {

                    this.state.spotsLeft = 5;

                    sessionStorage.setItem('spotsLeft', this.state.spotsLeft);

                } else {

                    this.state.spotsLeft = parseInt(spots);

                }

                this.updateScarcityBarText();


                // Simulate a spot being taken only once per session

                if (!sessionStorage.getItem('spotTaken') && this.state.spotsLeft > 1) {

                    setTimeout(() => {

                        this.state.spotsLeft--;

                        sessionStorage.setItem('spotsLeft', this.state.spotsLeft);

                        this.updateScarcityBarText();

                        

                        const strongEl = this.elements.scarcityBar.querySelector('strong');

                        if(strongEl) {

                            strongEl.classList.add('updated');

                            setTimeout(() => strongEl.classList.remove('updated'), 300);

                        }


                        this.elements.scarcityBar.classList.add('flash');

                        setTimeout(() => this.elements.scarcityBar.classList.remove('flash'), 500);


                        sessionStorage.setItem('spotTaken', 'true');

                    }, 10000 + Math.random() * 6000); // 10-16 seconds

                }

            },


            /**

             * Updates the text of the scarcity bar, ensuring it's always in sync.

             */

            updateScarcityBarText() {

                 if (this.state.spotsLeft <= 0) {

                    this.elements.scarcityBar.innerHTML = this.translations[this.state.currentLang].offer_expired;

                    return;

                }

                this.elements.scarcityBar.innerHTML = this.translations[this.state.currentLang].scarcity_bar_text.replace('{{spots_left}}', this.state.spotsLeft);

            }

        };


        // Run the application

        SeductionGodApp.init();


    </script>
