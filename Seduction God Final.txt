            const observer = new IntersectionObserver((entries) => {
                entries.forEach(entry => {
                    if (entry.isIntersecting) {
                        entry.target.classList.add('is-visible');
                    }
                });
            }, { threshold: 0.1 });
            document.querySelectorAll('.fade-in-section').forEach(el => observer.observe(el));

            // --- Daily 2-Hour Countdown Timer Logic ---
            // Creates a 2-hour scarcity window for each user, resetting daily.
            const timerDisplay = document.getElementById('countdown-timer');
            if (timerDisplay) {
                const priceBox = timerDisplay.closest('.price-box');
                let countdownEndTime;

                const now = new Date();
                const today = now.toDateString(); // A unique key for the current day.
                const storedData = JSON.parse(localStorage.getItem('seductionGodOffer'));

                if (storedData && storedData.day === today) {
                    // If the user has visited today, use the already-set expiration time.
                    countdownEndTime = storedData.endTime;
                } else {
                    // If it's the first visit of the day, set a new 2-hour timer.
                    countdownEndTime = now.getTime() + 2 * 60 * 60 * 1000; // 2 hours from now in milliseconds.
                    localStorage.setItem('seductionGodOffer', JSON.stringify({
                        endTime: countdownEndTime,
                        day: today
                    }));
                }

                const timerInterval = setInterval(() => {
                    const currentTime = new Date().getTime();
                    const distance = countdownEndTime - currentTime;

                    if (distance < 0) {
                        clearInterval(timerInterval);
                        priceBox.innerHTML = `<div class="limited-time" style="font-size: 1.5rem; color: #ff4444;">Η ΠΡΟΣΦΟΡΑ ΕΛΗΞΕ ΓΙΑ ΣΗΜΕΡΑ</div>`;
                        return;
                    }

                    const hours = Math.floor((distance % (1000 * 60 * 60 * 24)) / (1000 * 60 * 60));
                    const minutes = Math.floor((distance % (1000 * 60 * 60)) / (1000 * 60));
                    const seconds = Math.floor((distance % (1000 * 60)) / 1000);

                    // Format numbers to always have two digits (e.g., 09, 08, 07)
                    const fHours = hours < 10 ? "0" + hours : hours;
                    const fMinutes = minutes < 10 ? "0" + minutes : minutes;
                    const fSeconds = seconds < 10 ? "0" + seconds : seconds;

                    timerDisplay.textContent = `${fHours}:${fMinutes}:${fSeconds}`;

                }, 1000);
            }
        });
    </script>
